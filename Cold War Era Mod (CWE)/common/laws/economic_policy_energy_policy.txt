law_no_nuclear_energy = {
	group = lawgroup_energy_policy
	
	icon = "gfx/interface/icons/law_icons/laissez_faire.dds"
	
	#progressiveness = 0
	
	can_enact = {
	
		#Constitutional Laws system (Requires simple majority to amend constitutional laws)
		government_has_legislative_simple_majority = yes
		
	}
	
	on_activate = {
		custom_tooltip = {
			text = nuclear_energy_banned_destroy_tt
			every_scope_state = {
				if = {
					limit = {
						has_building = building_nuclear_power_plant
					}
					remove_building = building_nuclear_power_plant
				}
			}
		}
	}
	
	possible_political_movements = {
		#law_no_nuclear_energy
		law_nuclear_energy
	}
	
	#This law is not worth revolting over
	pop_support = { value = 0 }
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
	
}
