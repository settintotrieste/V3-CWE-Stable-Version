join_WTO_decision = {

	is_shown = {
		NOT = { has_modifier = WTO_member }
		
		#Date of establishment
		game_date >= 1995.1.1
		
		is_revolutionary = no
	}

	possible = {
		
		#NOT isolationist
		NOT = { has_law = law_type:law_isolationism }
		
		#Standard checks
		is_sufficiently_independent = yes
		is_at_war = no
		
		#THIS is not a pariah
		infamy < infamy_threshold:notorious
		
		#AI will only join the WTO if it wants to have Free Trade
		exists = ruler
		any_interest_group = {
			leader = { is_ruler = yes }
			law_stance = {
				law = law_type:law_free_trade
				value > neutral
			}
		}
		
	}
	
	when_taken = {
	
		#THIS joins the WTO
		add_modifier = { name = WTO_member }
		
		#THIS enacts free trade
		if = {
			limit = { NOT = { has_law = law_type:law_free_trade } }
			activate_law = law_type:law_free_trade
		}
		
	}

	ai_chance = {
		base = 0	
		
		modifier = {
		
			trigger = {
				
				#AI will only join the WTO if it wants to have Free Trade
				any_interest_group = {
					leader = { is_ruler = yes }
					law_stance = {
						law = law_type:law_free_trade
						value > neutral
					}
				}
				
			}
			
			add = 100
			
		}
		
	}
	
}

leave_WTO_decision = {

	is_shown = {
		has_modifier = WTO_member
	}

	possible = {
		
		#Standard checks
		is_sufficiently_independent = yes
		is_at_war = no
		
		#Have a ruler who hates free trade
		exists = ruler
		any_interest_group = {
			leader = { is_ruler = yes }
			law_stance = {
				law = law_type:law_free_trade
				value < neutral
			}
		}
		
	}
	
	when_taken = {
	
		#THIS joins the WTO
		remove_modifier = WTO_member
		
	}

	ai_chance = {
		base = 0	
		
		modifier = {
		
			trigger = {
				
				#AI will only leave the WTO if it hates Free Trade
				any_interest_group = {
					leader = { is_ruler = yes }
					law_stance = {
						law = law_type:law_free_trade
						value < disapprove
					}
				}
				
			}
			
			add = 100
			
		}
		
	}
	
}
