legal_union = {
	requires_approval = yes
	can_use_obligations = yes

	potential = {
		
		has_journal_entry = je_judiciary
		scope:target_country = { has_journal_entry = je_judiciary }
		
		#THIS is not decentralized
		NOT = { is_country_type = decentralized }
		scope:target_country = { NOT = { is_country_type = decentralized } }
		
		#THIS and Target are not revolutionary
		is_revolutionary = no
		scope:target_country = { is_revolutionary = no } 
		
		#THIS and Target are not subjects
		is_sufficiently_independent = yes
		scope:target_country = { 
			OR = { 
				is_sufficiently_independent = yes 
				is_direct_subject_of = ROOT
			}
		}
		
		#TARGET and THIS are not at war
		is_at_war = no
		scope:target_country = { is_at_war = no }
		
		#THIS is of a higher rank than the Target
		scope:target_country = { country_rank < root.country_rank }
		
		#TARGET is not already in a legal union
		scope:target_country = {
			NOT = { any_scope_diplomatic_pact = { is_diplomatic_action_type = legal_union } }
		}
		
	}

	possible = {
	
		#THIS is interested in the TARGET
		has_diplomatic_relevance = scope:target_country
		
		# THIS has good relations with TARGET
		custom_tooltip = {
			text = DIPLO_ACTION_RELATIONS_CORDIAL_OR_GREATER
			scope:target_country.relations:root >= relations_threshold:cordial
		}
		
		#Sufficient government similarity
		OR = {
		
			#THIS shares the same government type as TARGET
			custom_tooltip = {
				text = has_same_gov_type_tt
				OR = {
					AND = { is_a_democracy = yes scope:target_country = { is_a_democracy = yes } }
					AND = { is_a_communist_government = yes scope:target_country = { is_a_communist_government = yes } }
					AND = { is_a_nationalist_government = yes scope:target_country = { is_a_nationalist_government = yes } }
					AND = { is_a_theocratic_government = yes scope:target_country = { is_a_theocratic_government = yes } }
				}
			}
			
			#TARGET is a subject of THIS
			scope:target_country = { 
				is_direct_subject_of = ROOT
				OR = { 
					is_a_colonial_subject_gov = yes
					is_subject_type = subject_type_special_administrative_region
				}
			}
			
		}
		
	}
	
	accept_effect = {
	
		change_relations = {
			country = scope:target_country
			value = 25
		}
		
		#Harmonise judiciary of TARGET with THIS
		custom_tooltip = {
			text = legal_union_1.tt
			save_scope_as = legal_union_leader
			scope:target_country = { 
				save_scope_as = legal_union_member
				set_variable = { 
					name = judiciary_size
					value = scope:actor.var:judiciary_size
				}
				set_variable = {
					name = judiciary_progressiveness
					value = scope:actor.var:judiciary_progressiveness
				}
			}
		}
		
		#Harmonise laws of TARGET if TARGET is a confederal state of THIS
		if = {
			limit = {
				scope:target_country = { 
					is_direct_subject_of = scope:actor 
					is_subject_type = subject_type_confederal_state
				}
			}
			custom_tooltip = {
				text = harmonise_confederal_state_law.tt
				harmonise_confederal_state_law = yes
			}
			scope:target_country = { set_variable = THIS_in_confederal_legal_union }
		}
		
	}
	
	pact = {
		cost = 5
		is_two_sided_pact = yes
		
		break_effect = {
		
			change_relations = {
				country = scope:target_country
				value = -25
			}
			
			custom_tooltip = {
				text = legal_union_2.tt
				save_scope_as = legal_union_leader
				scope:target_country = { 
					save_scope_as = legal_union_member
				
					#Default is an even split
					set_variable = {
						name = judiciary_progressiveness
						value = 6
					}
					
					#Default Supreme Court size is 12
					set_variable = {
						name = judiciary_size
						value = 12
					}
					
				}
			}
			
			if = {
				limit = { 
					scope:target_country = { 
						is_direct_subject_of = scope:actor
						is_subject_type = subject_type_confederal_state
					} 
					has_variable = THIS_in_confederal_legal_union
				}
				remove_variable = THIS_in_confederal_legal_union
			}
			
		}
		
		target_can_break = {
			
			#Subjects cannot freely leave Legal Union
			scope:target_country = { NOT = { is_subject_of = scope:actor } }
			
		}
		
		should_auto_break = {
			
			NOT = { scope:target_country = { is_direct_subject_of = scope:actor NOT = { is_sufficiently_independent = yes } } }
			
			OR = {
			
				# THIS has bad relations with TARGET
				custom_tooltip = {
					text = DIPLO_ACTION_RELATIONS_COLD_OR_LESS
					scope:target_country.relations:root <= relations_threshold:cold
				}
			
				#Insufficient government similarity
				custom_tooltip = {
					text = has_same_gov_type_tt
					NOR = {
						AND = { is_a_democracy = yes scope:target_country = { is_some_kind_of_democracy = yes } }
						AND = { is_a_communist_government = yes scope:target_country = { is_a_communist_government = yes } }
						AND = { is_a_nationalist_government = yes scope:target_country = { is_a_nationalist_government = yes } }
						AND = { is_a_theocratic_government = yes scope:target_country = { is_a_theocratic_government = yes } }
					}
				}
				
				#THIS and TARGET no longer have independent courts
				has_law = law_type:law_kangaroo_law
				scope:target_country = { has_law = law_type:law_kangaroo_law }
				
				#TARGET is in a confederation but not a subject of THIS
				scope:target_country = { 
					is_subject_type = subject_type_confederal_state
					NOT = { is_direct_subject_of = scope:actor }
				}
			
			}
			
		}
		
		should_invalidate = {
			
			NOT = { scope:target_country = { is_direct_subject_of = scope:actor NOT = { is_sufficiently_independent = yes } } }
			
			OR = {
			
				NOT = { is_sufficiently_independent = yes }
				
				scope:target_country = { 
					NOT = { is_direct_subject_of = ROOT }
					NOT = { is_sufficiently_independent = yes }
				}
				
			}
			
		}
		
		monthly_effect = {
		
			#Harmonise judiciary of TARGET with THIS
			scope:target_country = { 
				set_variable = { 
					name = judiciary_size
					value = scope:actor.var:judiciary_size
				}
				set_variable = {
					name = judiciary_progressiveness
					value = scope:actor.var:judiciary_progressiveness
				}
			}
			
			#Harmonise laws of TARGET if TARGET is a confederal state of THIS
			if = {
				limit = {
					scope:target_country = { 
						is_direct_subject_of = scope:actor 
						is_subject_type = subject_type_confederal_state
					}
				}
				scope:target_country = { 
					custom_tooltip = {
						text = harmonise_confederal_state_law.tt
						harmonise_confederal_state_law = yes
					}
					set_variable = THIS_in_confederal_legal_union
				}
			}
		
		}
		
	}
	
	ai = {
		check_acceptance_for_will_propose = yes
		
		will_propose = { 
			country_rank >= rank_value:major_power
			OR = {
				AND = {
					infamy < infamy_threshold:infamous
					has_attitude = {
						who = scope:target_country
						attitude = protective
					}					
				}
				AND = {
					infamy < infamy_threshold:notorious
					has_attitude = {
						who = scope:target_country
						attitude = domineering
					}				
				}			
			}
		}
		
		propose_score = {
			value = 10
		}
		
		accept_score = {
			value = 0
	
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_BASE"
				value = -100
			}		

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RELATIONS"					
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:cordial
						"relations(scope:actor)" < relations_threshold:amicable
					}
					value = 10
				}					
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:amicable
						"relations(scope:actor)" < relations_threshold:friendly
					}
					value = 20
				}	
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:friendly
					}
					value = 40
				}				
			}
			
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_ATTITUDE"
				if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:actor
								attitude = protective
							}
							has_attitude = {
								who = scope:actor
								attitude = domineering
							}							
						}
					}
					value = 40
				}
				if = {
					limit = {
						has_attitude = {
							who = scope:actor
							attitude = genial
						}
					}
					value = 30
				}				
				if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:actor
								attitude = conciliatory
							}	
							has_attitude = {
								who = scope:actor
								attitude = cooperative
							}							
						}
					}
					value = 20
				}
				if = {
					limit = {
						has_attitude = {
							who = scope:actor
							attitude = cautious
						}
					}
					value = -50
				}
				if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:actor
								attitude = antagonistic
							}
							has_attitude = {
								who = scope:actor
								attitude = belligerent
							}
							has_attitude = {
								who = scope:actor
								attitude = rebellious
							}
						}
					}
					value = -1000
				}
			}
			
			add = {
				desc = "DIPLOMATIC_SUBJECT_ACCEPTANCE"					
				if = {
					limit = {
						scope:target_country = { 
							NOR = { 
								is_direct_subject_of = scope:actor
								is_subject_type = subject_type_special_administrative_region
							}
						}
					}
					value = -50
				}
				if = {
					limit = {
						scope:target_country = { 
							is_direct_subject_of = scope:actor
							is_subject_type = subject_type_special_administrative_region
						}
					}
					value = 20
				}		
				if = {
					limit = {
						scope:target_country = { 
							is_direct_subject_of = scope:actor
							is_a_colonial_subject_gov = yes
						}
					}
					value = 60
				}				
			}
			
			if = {
				limit = {
					"scope:actor.ai_ideological_opinion(root)" > 0
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_IDEOLOGICAL_OPINION"
					value = "scope:actor.ai_ideological_opinion(root)"
					multiply = 0.5
					max = 50
				}					
			}
			else_if = {
				limit = {
					"scope:actor.ai_ideological_opinion(root)" < 0
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_IDEOLOGICAL_OPINION_NEGATIVE"
					value = "scope:actor.ai_ideological_opinion(root)"
					multiply = 0.5
					min = -50
				}	
			}
			
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_INFAMY"
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:infamous
						"scope:actor.infamy" < infamy_threshold:notorious
					}
					value = -25
				}	
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:notorious
						"scope:actor.infamy" < infamy_threshold:pariah
					}
					value = -50
				}
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:pariah
					}
					value = -100
				}		
			}
		
		}
		
		check_acceptance_for_will_break = yes
		
		will_break = {
		
			NOT = {
				scope:target_country = { is_direct_subject_of = scope:actor }
			}
		
			scope:target_country = {
				OR = {
					
					AND = { 
						exists = ruler
						ruler = {
							OR = {
								has_ideology = ideology:ideology_autocratic
								has_ideology = ideology:ideology_power_hungry
							}
						}
					}
				
					has_attitude = {
						who = scope:actor
						attitude = rebellious
					}
					
				}
			}
			
		}
		
		use_obligation_chance = {
			value = 25
		}

		owe_obligation_chance = {
			value = 25
		}
		
	}
	
}
